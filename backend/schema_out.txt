
tbl_cp_mcountries:
  PK: row_id (int)
  AUTO_INCREMENT: row_id
  NOT NULL no default: country_id (int)
  NOT NULL no default: country_name (varchar(100))
  NOT NULL no default: country_code (varchar(5))

  NOT NULL no default: country_id (int)
